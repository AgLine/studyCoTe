peek()를 사용할때는 스택이 비어있지는 않은지 확인해야함

```
import java.util.Stack;

class Solution
{
    public int solution(String s)
    {
        Stack<Character> stack = new Stack<>();

        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            //스택이 비어있지않고 스택의 윗부분과 같다면 앞뒤로 같은 문자이므로 스택에서 삭제
            if(!stack.isEmpty() && stack.peek()==c){
                stack.pop();
            }else{//아니라면 현재의 값을 스택에 푸시
                stack.push(c);
            }
        }

        //for문이 끝나고 스택이 비어있다면 모두 짝이 있는 상태인거임
        if(stack.isEmpty()){
            return 1;
        }
        
        return 0;
    }
}
```
